45 static cv::Mat translate(cv::Mat input, double fromMin, double fromMax, double toMin, double toMax){
46 //Figure out how 'wide' each range is
47 double fromSpan = fromMax - fromMin;
48 double toSpan = toMax - toMin;
49
50 //Convert the left range into a 0-1 range (float)
51 cv::Mat valueScaled = input - fromMin / fromSpan;
52
53 //Convert the 0-1 range into a value in the right range.
54 cv::Mat translated = toMin + (valueScaled * toSpan);
55 return translated;
56 }

114 static cv::Mat genWImage(cv::Mat disp, cv::Mat Q, float downsample_factor = 1.0){
115 float q32 = Q.at<float>(3, 2);
116 float q33 = Q.at<float>(3, 3);
117 float d, w;
118 cv::Mat wImage = cv::Mat(disp.rows, disp.cols, CV_32FC3, cv::Scalar(0));
119
120 float downsample_rate = 1 / downsample_factor;
121
122 for (int i = 0; i < disp.rows; i++)
123 {
124 for (int j = 0; j < disp.cols; j++)
125 {
126 d = disp.at<float>(i, j);
127 if (d != 0)
128 {
129 w = ((d * downsample_rate) * q32) + q33;
130 wImage.at<float>(i,j) = w;
131 }
132 }
133 }
134
135 return wImage;

306 {
307 for (int j = 0; j < disparity_thresh.cols; j++)
308 {
309 float d = disparity_thresh.at<float>(i, j);
310 if (d > max_disp || d < min_disp){
311 disparity_thresh.at<float>(i, j) = 0;
312 }
313 }
314 }
315
316 // normalise disparity
317 cv::Mat normDisp;
318 normaliseDisparity(disparity_thresh, normDisp);
319
320 for (int i = 0; i < disparity_thresh.rows; i++)
321 {
322 for (int j = 0; j < disparity_thresh.cols; j++)
323 {
324 if (normDisp.at<uchar>(i, j) == 255 || normDisp.at<uchar>(i, j) == 0){
325 disparity_thresh.at<float>(i, j) = 0;
326 }
327 }
328 }
329

333 static float getHFOVFromQ(cv::Mat Q)
334 {
335 // Get horitonzal Field Of View of camera from Q matrix
336 // Useful for reconstructing XYZ depth from Z only depth
337 // Can be embedded in top left corner pixel of RGBD image to simplify reconstruction
338
339 // Q Matrix has the format
340 // [ 1 0 0 -cx ]
341 // [ 0 1 0 -cy ]
342 // [ 0 0 0 fx ]
343 // [ 0 0 -1/Tx (cx-cxr)/Tx ]
344
345 // Cx and Cy are principal point of the left camera (in pixels so half the resolution)
346 // Fx is horizontal focal length of left camera (in pixels)
347
348 // Extract focal length and principal point from Q
349 float fx = Q.at<float>(2, 3);
350 float cx = -Q.at<float>(0, 3);
351
352 // Calculate FOV
353 // See here for explaination of the geometric reasoning
354 // https://photo.stackexchange.com/questions/41273/how-to-calculate-the-fov-in-degrees-from-focal-length-or-distance
355 float fov_x = 2 * atan(cx / fx);
356
357 // Return horizonal FOV in radians
358 return fov_x;
359 }

361 static void disparity2CVPointCloud(cv::Mat disparity, cv::Mat image, cv::Mat Q, cv::Mat &depth, cv::Mat &depthColors, float max_z = 10000){
362 cv::Mat disparity16;
363 disparity.copyTo(disparity16);
364
365 if (Q.empty() || disparity16.empty()) {
366 return;
367 }
368
369 depth = cv::Mat::zeros(cv::Size(3, 1), CV_32FC1);
370 depthColors = cv::Mat::zeros(cv::Size(3, 1), CV_8UC1);
371
372 float wz = Q.at<float>(2, 3);
373 float q03 = Q.at<float>(0, 3);
374 float q13 = Q.at<float>(1, 3);
375 float q32 = Q.at<float>(3, 2);
376 float q33 = Q.at<float>(3, 3);
377 float w, d;
378 uchar intensity;
379
380 float xyz[3] = {0,0,0};
381 uchar rgb[3] = {0,0,0};
382
383 for (int i = 0; i < disparity16.rows; i++)
384 {
385 for (int j = 0; j < disparity16.cols; j++)
386 {
387 d = disparity16.at<float>(i, j);
388 if (d != 0)
389 {
390 w = (d * q32) + q33;
391 xyz[0] = (j + q03) / w;
392 xyz[1] = (i + q13) / w;
393 xyz[2] = wz / w;
394
395 if (w > 0 && xyz[2] > 0 && xyz[2] < max_z){ // negative W or Z which is not possible (behind camera)
396 if (image.type() == CV_8UC1){
397 // MONO8
398 intensity = image.at<uchar>(i,j);
399 rgb[0] = intensity;
400 rgb[1] = intensity;
401 rgb[2] = intensity;
402 } else if (image.type() == CV_8UC3){
403 // BGR8
404 rgb[2] = image.at<cv::Vec3b>(i,j)[0];
405 rgb[1] = image.at<cv::Vec3b>(i,j)[1];
406 rgb[0] = image.at<cv::Vec3b>(i,j)[2];
407 } else {
408 rgb[0] = 0;
409 rgb[1] = 0;
410 rgb[2] = 0;
411 //qDebug() << "Invalid image type. MUST be CV_8UC1 or CV_8UC3";
412 }
413
414 cv::Mat point = cv::Mat(1, 3, CV_32FC1, &xyz);
415 cv::Mat color = cv::Mat(1, 3, CV_8UC1, &rgb);
416 //std::cout << "(" << rgb[0] << "," << rgb[1] << "," << rgb[2] << ")" << std::endl;
417 depth.push_back(point);
418 depthColors.push_back(color);
419 }
420 }
421 }

468 static void disparity2colormap(cv::Mat inDisparity, cv::Mat Q, cv::Mat &outColormap, bool isFiltered=false){
469 // threshold value disparity
470 cv::Mat validDisp;
471 if (!isFiltered){
472 removeInvalidDisparity(inDisparity, Q, validDisp);
473 } else {
474 validDisp = inDisparity.clone();
475 }
476

478 validDisp.convertTo(floatDisp, CV_32F);
479
480 // normalise disparity

483
484 // apply colormap

486 cv::Mat colormap;

493 colormap.at<cv::Vec3b>(y, x)[0] = 0;
494 colormap.at<cv::Vec3b>(y, x)[1] = 0;
495 colormap.at<cv::Vec3b>(y, x)[2] = 0;
496 }
497 }
498 }
