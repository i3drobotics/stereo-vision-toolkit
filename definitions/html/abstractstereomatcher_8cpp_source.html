20void AbstractStereoMatcher::convertImages(cv::Mat left_img, cv::Mat right_img, cv::Mat& left_bgr_conv_img, cv::Mat& left_conv_img, cv::Mat& right_conv_img) {
21 cv::Mat right_tmp, left_tmp, left_bgr_tmp;
22
23 if (left_img.type() == CV_8UC1){
24 left_img.copyTo(left_bgr_tmp);
25 left_img.copyTo(left_tmp);
26 } else if (left_img.type() == CV_8UC3){
27 left_img.copyTo(left_bgr_tmp);
28 cv::cvtColor(left_bgr_tmp,left_tmp,cv::COLOR_BGR2GRAY);
29 left_tmp.convertTo(left_tmp, CV_8UC1);
30 } else {
31 std::cerr << "Unknown image type in left image for stereo matching: " << left_img.type() << ". Expected CV_8U1 or CV_8UC3" << std::endl;
32 left_img.copyTo(left_bgr_tmp);
33 left_img.copyTo(left_tmp);
34 }
35
36 if (right_img.type() == CV_8UC1){
37 right_img.copyTo(right_tmp);
38 } else if (right_img.type() == CV_8UC3){
39 cv::cvtColor(right_img,right_tmp,cv::COLOR_BGR2GRAY);
40 right_tmp.convertTo(right_tmp, CV_8UC1);
41 } else {
42 std::cerr << "Unknown image type in right image for stereo matching: " << right_img.type() << ". Expected CV_8U1 or CV_8UC3" << std::endl;
43 right_img.copyTo(right_tmp);
44 }
45
46 if (downsample_factor != 1){
47 cv::resize(right_tmp,right_tmp,cv::Size(),downsample_factor,downsample_factor);
48 cv::resize(left_bgr_tmp,left_bgr_tmp,cv::Size(),downsample_factor,downsample_factor);
49 cv::resize(left_tmp,left_tmp,cv::Size(),downsample_factor,downsample_factor);
50 }
51
52 left_conv_img = left_tmp.clone();
53 left_bgr_conv_img = left_bgr_tmp.clone();
54 right_conv_img = right_tmp.clone();
55}
