73void DisparityViewer::updateDisparity() {
74 cv::Mat disparity, disparity_thresh;
75 matcher->getDisparity16(disparity);
76 disparity.copyTo(disparity_thresh);
77 //double d_error = 16 * matcher->getErrorDisparity();
78
79 /*
80
81 double min_disp = 10000;
82 double max_disp = 0;
83 int min_i = 0;
84 int max_i = 0;
85 int min_j = 0;
86 int max_j = 0;
87
88 double min_depth = 10000;
89 double max_depth = 0;
90
91 double d_error = 16 * matcher->getErrorDisparity();
92
93 cv::Matx44d _Q;
94 Q.convertTo(_Q, CV_64F);
95
96 for (int i = 0; i < disparity.rows; i++)
97 {
98 for (int j = 0; j < disparity.cols; j++)
99 {
100 float d = disparity.at<float>(i, j);
101 if (d < 10000 && d != d_error){
102 float d = disparity.at<float>(i, j);
103 if (d < min_disp){
104 //TODO find out why issue with disp < ~3 causing negative w and so negative z
105 if (CVSupport::genZ(_Q,i,j,d) > 0){
106 min_disp = d;
107 min_i = i;
108 min_j = j;
109 }
110 }
111 if (d > max_disp){
112 max_disp = d;
113 max_i = i;
114 max_j = j;
115 }
116 }
117 }
118 }
119
120 min_depth = (double)CVSupport::genZ(_Q,(double)max_i,(double)max_j,(double)max_disp);
121 max_depth = (double)CVSupport::genZ(_Q,(double)min_i,(double)min_j,(double)min_disp);
122 */
123
124 double min_disp, max_disp, min_depth, max_depth;
125 CVSupport::getMinMaxDisparity(disparity_thresh,Q,min_disp,max_disp);
126 CVSupport::getMinMaxDepth(disparity_thresh,Q,min_depth,max_depth);
127 min_depth_ = min_depth * this->downsample_factor;
128 max_depth_ = max_depth * this->downsample_factor;
129
130 //use depth rather than disparity for more user readable results
131 double range_disp = max_disp - min_disp;
132 double range_depth = max_depth - min_depth;
133
134 ui->minDisparityLabel->setText(
135 QString("%1 px").arg(QString::number(min_disp,'g', 4)));
136 ui->disparityRangeLabel->setText(
137 QString("%1 px").arg(QString::number(range_disp,'g', 4)));
138 ui->maxDisparityLabel->setText(
139 QString("%1 px").arg(QString::number(max_disp,'g', 4)));
140
141 ui->minDepthLabel->setText(
142 QString("%1 m").arg(QString::number(min_depth,'g', 4)));
143 ui->depthRangeLabel->setText(
144 QString("%1 m").arg(QString::number(range_depth,'g', 4)));
145 ui->maxDepthLabel->setText(
146 QString("%1 m").arg(QString::number(max_depth,'g', 4)));
147
148 cv::Mat disparity_vis;
149 CVSupport::disparity2colormap(disparity,Q,disparity_vis);
150
151 cv::cvtColor(disparity_vis, disparity_vis, cv::COLOR_BGR2RGB);
152
153 QImage dmap(disparity_vis.data, disparity_vis.cols, disparity_vis.rows,
154 QImage::Format_RGB888);
155
156 QPixmap pmap_disparity = QPixmap::fromImage(dmap);
157
158 if(!pmap_disparity.isNull()){
159 emit newDisparity(
160 pmap_disparity.scaled(viewer->size(), Qt::KeepAspectRatio));
161 }
162
163 processing_disparity = false;
164}
