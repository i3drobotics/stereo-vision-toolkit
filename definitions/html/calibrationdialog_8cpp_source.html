61std::vector<cv::Point2f> CalibrationDialog::getBoardOutsideCorners(std::vector<cv::Point2f> corners){
62
63 int xdim = ui->patternColSpinBox->value();
64 int ydim = ui->patternRowSpinBox->value();
65
66 std::vector<cv::Point2f> outside_corners;
67 int top_left_index = getCornerIndex(1,1);
68 int top_right_index = getCornerIndex(1,ydim);
69 int bot_left_index = getCornerIndex(xdim,1);
70 int bot_right_index = getCornerIndex(xdim,ydim);
71
72 cv::Point2f top_left = corners.at(top_left_index);
73 cv::Point2f top_right = corners.at(top_right_index);
74 cv::Point2f bot_left = corners.at(bot_left_index);
75 cv::Point2f bot_right = corners.at(bot_right_index);
76
77 outside_corners.push_back(top_left);
78 outside_corners.push_back(top_right);
79 outside_corners.push_back(bot_left);
80 outside_corners.push_back(bot_right);
81
82 return outside_corners;

103bool CalibrationDialog::checkImages(cv::Mat left, cv::Mat right){
104 // check images should be added to list
105 std::vector<cv::Point2f> left_corners;
106 // downsample image
107 cv::Mat left_d = downsampleImage(left);
108 cv::Mat right_d = downsampleImage(right);
109 if (findCorners(left_d,left_corners)){
110 std::vector<cv::Point2f> right_corners;
111 if (findCorners(right_d,right_corners)){
112 //Add images to database only if it's sufficiently different from any previous sample.
113 cv::RotatedRect left_rect = getBoardRect(left_corners);
114 cv::RotatedRect right_rect = getBoardRect(right_corners);
115 std::vector<cv::Point2f> left_outer_corners = getBoardOutsideCorners(left_corners);
116 std::vector<cv::Point2f> right_outer_corners = getBoardOutsideCorners(right_corners);
117 bool skew_diff = checkSkewDiff(left_outer_corners,right_outer_corners);
118 bool size_diff = checkSizeDiff(left_rect,right_rect);
119 bool XY_diff = checkXYDiff(left_rect,right_rect);
120 updateProgress();
121 if (XY_diff || skew_diff || size_diff){
122 return true;
123 }
124 }
125 }
126 return false;

166bool CalibrationDialog::checkSkewDiff(std::vector<cv::Point2f> left_outer_corners, std::vector<cv::Point2f> right_outer_corners){
167
168 bool isDifferent = false;
169
170 cv::Point2f l_topLeft = left_outer_corners.at(0);
171 cv::Point2f l_topRight = left_outer_corners.at(1);
172 cv::Point2f l_bottomLeft = left_outer_corners.at(2);
173 cv::Point2f l_bottomRight = left_outer_corners.at(3);
174
175 cv::Point2f r_topLeft = right_outer_corners.at(0);
176 cv::Point2f r_topRight = right_outer_corners.at(1);
177 cv::Point2f r_bottomLeft = right_outer_corners.at(2);
178 cv::Point2f r_bottomRight = right_outer_corners.at(3);
179
180 double l_top_length = cv::norm(l_topRight-l_topLeft);
181 double l_bottom_length = cv::norm(l_bottomRight-l_bottomLeft);
182 double l_right_length = cv::norm(l_bottomRight-l_topRight);
183 double l_left_length = cv::norm(l_bottomLeft-l_topLeft);
184
185 double l_y_skew = l_top_length/l_bottom_length;
186 double l_x_skew = l_left_length/l_right_length;
187
188 double r_top_length = cv::norm(r_topRight-r_topLeft);
189 double r_bottom_length = cv::norm(r_bottomRight-r_bottomLeft);
190 double r_right_length = cv::norm(r_bottomRight-r_topRight);
191 double r_left_length = cv::norm(r_bottomLeft-r_topLeft);
192
193 double r_y_skew = r_top_length/r_bottom_length;
194 double r_x_skew = r_left_length/r_right_length;
195
196 double c_min_x_skew,c_max_x_skew, c_min_y_skew, c_max_y_skew;
197 if (l_x_skew < r_x_skew){
198 c_max_x_skew = r_x_skew;
199 c_min_x_skew = l_x_skew;
200 } else {
201 c_max_x_skew = l_x_skew;
202 c_min_x_skew = r_x_skew;
203 }
204 if (l_y_skew < r_y_skew){
205 c_max_y_skew = r_y_skew;
206 c_min_y_skew = l_y_skew;
207 } else {
208 c_max_y_skew = l_y_skew;
209 c_min_y_skew = r_y_skew;
210 }
211
212 if (min_x_skew == -1){
213 min_x_skew = c_min_x_skew;
214 isDifferent = true;
215 } else if (c_min_x_skew < (min_x_skew)-skew_min_change){
216 min_x_skew = c_min_x_skew;
217 isDifferent = true;
218 }
219 if (min_y_skew == -1){
220 min_y_skew = c_min_y_skew;
221 isDifferent = true;
222 } else if (c_min_y_skew < (min_y_skew)-skew_min_change){
223 min_y_skew = c_min_y_skew;
224 isDifferent = true;
225 }
226 if (max_x_skew == -1){
227 max_x_skew = c_max_x_skew;
228 isDifferent = true;
229 } else if (c_max_x_skew > (max_x_skew)+skew_min_change){
230 max_x_skew = c_max_x_skew;
231 isDifferent = true;
232 }
233 if (max_y_skew == -1){
234 max_y_skew = c_max_y_skew;
235 isDifferent = true;
236 } else if (c_max_y_skew > (max_y_skew)+skew_min_change){
237 max_y_skew = c_max_y_skew;
238 isDifferent = true;
239 }
240
241 return isDifferent;

244bool CalibrationDialog::checkXYDiff(cv::RotatedRect left_rot_rect, cv::RotatedRect right_rot_rect){
245
246 cv::Rect left_rect = left_rot_rect.boundingRect();
247 cv::Rect right_rect = right_rot_rect.boundingRect();
248
249 cv::Point left_minVal = left_rect.tl();
250 cv::Point right_minVal = right_rect.tl();
251
252 cv::Point left_maxVal = left_rect.br() - cv::Point(1, 1);
253 cv::Point right_maxVal = right_rect.br() - cv::Point(1, 1);
254
255 int c_x_lowest,c_y_lowest,c_x_highest,c_y_highest;
256
257 if (left_minVal.x < right_minVal.x){
258 c_x_lowest = left_minVal.x;
259 } else {
260 c_x_lowest = right_minVal.x;
261 }
262
263 if (left_minVal.y < right_minVal.y){
264 c_y_lowest = left_minVal.y;
265 } else {
266 c_y_lowest = right_minVal.y;
267 }
268
269 if (left_maxVal.x > right_maxVal.x){
270 c_x_highest = left_maxVal.x;
271 } else {
272 c_x_highest = right_maxVal.x;
273 }
274
275 if (left_maxVal.y > right_maxVal.y){
276 c_y_highest = left_maxVal.y;
277 } else {
278 c_y_highest = right_maxVal.y;
279 }
280
281 bool isDifferent = false;
282
283 if (x_lowest == -1){
284 x_lowest = c_x_lowest;
285 isDifferent = true;
286 } else if (c_x_lowest < (x_lowest)-p_min_change){
287 x_lowest = c_x_lowest;
288 isDifferent = true;
289 }
290 if (y_lowest == -1) {
291 y_lowest = c_y_lowest;
292 isDifferent = true;
293 } else if ( c_y_lowest < (y_lowest)-p_min_change){
294 y_lowest = c_y_lowest;
295 isDifferent = true;
296 }
297 if (x_highest == -1){
298 x_highest = c_x_highest;
299 isDifferent = true;
300 } else if (c_x_highest > (x_highest)+p_min_change){
301 x_highest = c_x_highest;
302 isDifferent = true;
303 }
304 if (y_highest == -1){
305 y_highest = c_y_highest;
306 isDifferent = true;
307 } else if (c_y_highest > (y_highest)+p_min_change){
308 y_highest = c_y_highest;
309 isDifferent = true;
310 }
311
312 return isDifferent;

394void CalibrationDialog::updateProgress(){
395 int x_range, y_range,size_range;
396 double x_skew_range,y_skew_range;
397 if (x_highest == -1 || x_lowest == -1){
398 x_range = 0;
399 } else {
400 x_range = x_highest - x_lowest;
401 }
402 if (y_highest == -1 || y_lowest == -1){
403 y_range = 0;
404 } else {
405 y_range = y_highest - y_lowest;
406 }
407 if (max_size == -1 || min_size == -1){
408 size_range = 0;
409 } else {
410 size_range = max_size - min_size;
411 }
412 if (max_x_skew == -1 || min_x_skew == -1){
413 x_skew_range = 0;
414 } else {
415 x_skew_range = max_x_skew - min_x_skew;
416 }
417 if (max_y_skew == -1 || min_y_skew == -1){
418 y_skew_range = 0;
419 } else {
420 y_skew_range = max_y_skew - min_y_skew;
421 }
422
423 qDebug() << "X range: " << x_range << " (" << x_lowest << " - " << x_highest << ")";
424 qDebug() << "Y range: " << y_range << " (" << y_lowest << " - " << y_highest << ")";
425 qDebug() << "Size range: " << size_range << " (" << min_size << " - " << max_size << ")";
426 qDebug() << "X Skew range: " << x_skew_range << " (" << min_x_skew << " - " << max_x_skew << ")";
427 qDebug() << "X Skew range: " << y_skew_range << " (" << min_y_skew << " - " << max_y_skew << ")";
428
429 int x_goal = image_width*0.6;
430 int y_goal = image_height*0.6;
431 int size_goal = ((image_height*image_width)/2)*0.6;
432 double x_skew_goal = 0.5;
433 double y_skew_goal = 0.3;
434
435 int x_complete = ((float)(x_range*m_scale)/(float)x_goal)*100;
436 int y_complete = ((float)(y_range*m_scale)/(float)y_goal)*100;
437 int size_complete = ((float)(size_range*(m_scale*m_scale))/(float)size_goal)*100;
438 int x_skew_complete = ((float)x_skew_range/(float)x_skew_goal)*100;
439 int y_skew_complete = ((float)y_skew_range/(float)y_skew_goal)*100;
440
441 // calculate total complete from other progress status
442 int total_complete = ((float)(x_complete+y_complete+size_complete+x_skew_complete+y_skew_complete)/(float)500)*100;
443
444 // output progress to progress bars
445 ui->progressBarX->setValue(x_complete);
446 ui->progressBarY->setValue(y_complete);
447 ui->progressBarSize->setValue(size_complete);
448 ui->progressBarXSkew->setValue(x_skew_complete);
449 ui->progressBarYSkew->setValue(y_skew_complete);
450
451 ui->progressBarComplete->setValue(total_complete);
452}
