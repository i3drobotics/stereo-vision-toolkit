8bool StereoCameraTIS::openCamera(){
9 if (isConnected()){
10 closeCamera();
11 }
12 auto left_camera_tmp = new CameraImagingSource();
13 auto right_camera_tmp = new CameraImagingSource();
14
15 std::string left_serial = stereoCameraSerialInfo_.left_camera_serial;
16 std::string right_serial = stereoCameraSerialInfo_.right_camera_serial;
17
18 if (left_camera_tmp->open(left_serial) && right_camera_tmp->open(right_serial)) {
19 connected = setCameras(stereoCameraSettings_,left_camera_tmp,right_camera_tmp,new Listener(), new Listener());
20 // TODO remove these when single capture works
21 left_camera->startCapture();
22 right_camera->startCapture();
23 } else {
24 connected = false;
25 }
26 return connected;

93std::vector<AbstractStereoCamera::StereoCameraSerialInfo> StereoCameraTIS::listSystemsQuick(DShowLib::Grabber* handle){
94 std::vector<AbstractStereoCamera::StereoCameraSerialInfo> known_serial_infos = loadSerials(AbstractStereoCamera::CAMERA_TYPE_TIS);
95 std::vector<AbstractStereoCamera::StereoCameraSerialInfo> connected_serial_infos;
96
97 /* Get number of cameras */
98 auto devices = handle->getAvailableVideoCaptureDevices();
99
100 for (auto& known_serial_info : known_serial_infos) {
101 bool left_found = false;
102 bool right_found = false;
103 qint64 left_serial;
104 qint64 right_serial;
105 //find left
106 for (auto& device : *devices) {
107 device.getSerialNumber(left_serial);
108 if (QString::number(left_serial).toStdString() == known_serial_info.left_camera_serial){
109 left_found = true;
110 break;
111 }
112 }
113 if (left_found){
114 //find right
115 for (auto& device : *devices) {
116 device.getSerialNumber(right_serial);
117 if (QString::number(right_serial).toStdString() == known_serial_info.right_camera_serial){
118 right_found = true;
119 break;
120 }
121 }
122 }
123 if (left_found && right_found){ //only add if both cameras found
124 connected_serial_infos.push_back(known_serial_info);
125 }
126 }
127 return connected_serial_infos;
128}

179QList<qint64> StereoCameraTIS::load_serials(QString filename) {
180 QList<qint64> serials;
181
182 QFile inputFile(filename);
183 if (inputFile.open(QIODevice::ReadOnly)) {
184 QTextStream in(&inputFile);
185 while (!in.atEnd()) {
186 QStringList line = in.readLine().split(' ');
187
188 if(line.size() == 3){
189 qint64 serial = line.at(0).toLongLong();
190 int width = line.at(1).toInt();
191 int height = line.at(2).toInt();
192
193 if (serial) {
194 serials.append(serial);
195 widths.append(width);
196 heights.append(height);
197 qDebug() << "Looking for Phobos camera serial " << (qint64)serial;
198 }
199 }
200 }
201 inputFile.close();
202 } else {
203 qDebug() << "Couldn't open Phobos camera serials file " << filename;
204 }
205
206 return serials;
207}

209void StereoCameraTIS::setup_cameras(AbstractStereoCamera::StereoCameraSettings inital_camera_settings){
210
211 QThread* left_camera_thread = new QThread();
212 left_camera->assignThread(left_camera_thread);
213
214 QThread* right_camera_thread = new QThread();
215 right_camera->assignThread(right_camera_thread);
216
217 image_height = left_camera->height;
218 image_width = left_camera->width;
219 image_bitdepth = 1; //TODO get bit depth
220
221 emit update_size(image_width, image_height, image_bitdepth);
222
223 double exposure = inital_camera_settings.exposure;
224 int gain = inital_camera_settings.gain;
225 int fps = inital_camera_settings.fps;
226 bool trigger;
227 if (inital_camera_settings.trigger == 1){
228 trigger = true;
229 } else {
230 trigger = false;
231 }
232 bool autoExpose;
233 if (inital_camera_settings.autoExpose == 1){
234 autoExpose = true;
235 } else {
236 autoExpose = false;
237 }
238 bool autoGain;
239 if (inital_camera_settings.autoGain == 1){
240 autoGain = true;
241 } else {
242 autoGain = false;
243 }
244
245 left_camera->setup(image_width, image_height, fps);
246 right_camera->setup(image_width, image_height, fps);
247
248 left_raw.create(getSize(), CV_8UC1);
249 right_raw.create(getSize(), CV_8UC1);
250
251 left_listener->setOutputBuffer(left_raw.data);
252 right_listener->setOutputBuffer(right_raw.data);
253
254 left_camera->setListener((DShowLib::GrabberListener*) left_listener);
255 right_camera->setListener((DShowLib::GrabberListener*) right_listener);
256
257 assert(left_camera->getListener() != nullptr);
258 assert(right_camera->getListener() != nullptr);
259
260 connect((Listener*) left_camera->getListener(), SIGNAL(grabbed(void*)), this, SLOT(leftCaptured()));
261 connect((Listener*) right_camera->getListener(), SIGNAL(grabbed(void*)), this, SLOT(rightCaptured()));
262
263 connect(left_camera, SIGNAL(grabError()), this, SLOT(leftGrabFailed()));
264 connect(right_camera, SIGNAL(grabError()), this, SLOT(rightGrabFailed()));
265
266 connect(left_listener, SIGNAL(deviceDisconnected(void)), this, SLOT(closeCamera(void)));
267 connect(right_listener, SIGNAL(deviceDisconnected(void)), this, SLOT(closeCamera(void)));
268
269 connect(this, SIGNAL(start_capture(void)), left_camera, SLOT(startCapture(void)));
270 connect(this, SIGNAL(start_capture(void)), right_camera, SLOT(startCapture(void)));
271
272 connect(this, SIGNAL(stop_capture(void)), left_camera, SLOT(stopCapture(void)));
273 connect(this, SIGNAL(stop_capture(void)), right_camera, SLOT(stopCapture(void)));
274
275 connect(this, SIGNAL(stereo_grab(void)), left_camera, SLOT(grabImage(void)));
276 connect(this, SIGNAL(stereo_grab(void)), right_camera, SLOT(grabImage(void)));
277
278 enableTrigger(trigger);
279 enableAutoExposure(autoExpose);
280 setExposure(exposure);
281 enableAutoGain(autoGain);
282 setGain(gain);
283
284 connected = true;
285}
