120std::vector<AbstractStereoCamera::StereoCameraSerialInfo> AbstractStereoCamera::loadSerials(AbstractStereoCamera::StereoCameraType camera_type, std::string filename){
121 std::vector<StereoCameraSerialInfo> serials;
122 QString qFilname = QString::fromStdString(filename);
123 QFile inputFile(qFilname);
124 if (inputFile.open(QIODevice::ReadOnly)) {
125 QTextStream in(&inputFile);
126 while (!in.atEnd()) {
127 QStringList line = in.readLine().split(' ');
128
129 if(line.size() == 4 || line.size() == 5){ //TODO change this back to only 4 when excel that generates the serials is fixed
130 QString cam_type = line.at(0);
131 std::string file_cam_type_str = cam_type.toStdString();
132 StereoCameraType file_cam_type = String2StereoCameraType(file_cam_type_str);
133 if (file_cam_type == camera_type){
134 QString left_camera_serial = line.at(1);
135 QString right_camera_serial = line.at(2);
136 QString i3dr_serial = line.at(3);
137 AbstractStereoCamera::StereoCameraSerialInfo csi;
138 csi.camera_type = camera_type;
139 csi.left_camera_serial = left_camera_serial.toStdString();
140 csi.right_camera_serial = right_camera_serial.toStdString();
141 csi.i3dr_serial = i3dr_serial.toStdString();
142 serials.push_back(csi);
143 }
144 }
145 }
146 inputFile.close();
147 } else {
148 qDebug() << "Couldn't open i3dr camera serials file " << qFilname;
149 }
150 return serials;

189void AbstractStereoCamera::saveRGBD(QString filename, bool enable_16bit, float scale_16bit){
190 cv::Mat rgbd, color, depth, Q, depth_z, depth_split[3];;
191 getLeftMatchImage(color);
192 getDepth(depth);
193 getQ(Q);
194
195 // extract Z only channel from depth (xyz) image
196 cv::split(depth, depth_split);
197 depth_z = depth_split[2];
198 // get horizontal fov from Q matrix
199 float hfov = CVSupport::getHFOVFromQ(Q);
200 // embed horizontal fov in top left pixel of z only depth image to simplify reconstruction
201 depth_z.at<float>(0,0) = (float)hfov;
202 if (!depth_z.empty() && !color.empty()){
203 // convert image to rgb
204 if (color.type() == CV_8UC1){
205 cvtColor(color,color,cv::COLOR_GRAY2RGB);
206 } else if (color.type() == CV_8UC3){
207 cvtColor(color,color,cv::COLOR_BGR2RGB);
208 } else {
209 std::cerr << "Invalid image type for saving RGBD" << std::endl;
210 }
211 // Create RGBD image from color image and z only depth image
212 if (enable_16bit){
213 rgbd = CVSupport::createRGBD16(color,depth_z,scale_16bit,false);
214 } else {
215 rgbd = CVSupport::createRGBD32(color,depth_z);
216 }
217 qDebug() << "Saving rgbd image...";
218 cv::imwrite(filename.toStdString(), rgbd);
219 } else {
220 std::cerr << "Depth image or camera image is empty." << std::endl;
221 return;
222 }
223}

563 directory + "/right_calibration.xml",

565 qDebug() << "Couldn't load camera calibration";

569 QCoreApplication::processEvents();
570
571 if (load_rectification){
572 if (!loadXMLRectificationMaps(directory + "/left_rectification.xml", directory + "/right_rectification.xml")){
573 rectification_valid = false;
574 qDebug() << "Couldn't find rectification maps";
575 return false;

580 cal_image_height != image_height || rectmapy_r.rows != image_height) {
581 rectification_valid = false;
582 qDebug() << "Image size doesn't match rectification maps";
583 qDebug() << "Cal image size: " << cal_image_width << "," << cal_image_height;
584 qDebug() << "Rectmap size: " << rectmapx_r.cols << "," << rectmapy_r.rows;
585 qDebug() << "Camera image size: " << image_width << "," << image_height;
586
587 return false;
588 }
589 }
590 QCoreApplication::processEvents();
591
592 rectification_valid = true;
593
594 return true;
