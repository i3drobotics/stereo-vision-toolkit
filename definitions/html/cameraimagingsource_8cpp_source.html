61void CameraImagingSource::setup(int width, int height, int fps) {
62 setVideoFormat(width, height);
63
64 this->width = handle.getVideoFormat().getSize().cx;
65 this->height = handle.getVideoFormat().getSize().cy;
66
67 frame_sink = DShowLib::FrameHandlerSink::create(
68 DShowLib::FrameTypeInfoArray(DShowLib::eY800), 5);
69 frame_sink->setSnapMode(true);
70
71 if (fps > 0) {
72 if (!handle.setFPS(fps)) {
73 debugMessage("Failed to set framerate");
74 }
75 } else {
76 auto maximum_framerate = handle.getCurrentMaxAvailableFPS();
77 if (!handle.setFPS(maximum_framerate)) {
78 debugMessage("Failed to set framerate");
79 }
80 }
81}

172void CameraImagingSource::enableAutoExposure(bool enable){
173 DShowLib::tIVCDPropertyItemsPtr pItems = handle.getAvailableVCDProperties();
174 if( pItems != 0 )
175 {
176 // Try to find the exposure item
177 DShowLib::tIVCDPropertyItemPtr pExposureItem = pItems->findItem( DShowLib::VCDID_Exposure );
178
179 DShowLib::tIVCDSwitchPropertyPtr m_pExposureAuto;
180
181 DShowLib::tIVCDPropertyElementPtr pExposureAutoElement = pExposureItem->findElement( DShowLib::VCDElement_Auto );
182
183 // If an auto element exists, try to acquire a switch interface
184 if( pExposureAutoElement != 0 )
185 {
186 pExposureAutoElement->getInterfacePtr( m_pExposureAuto );
187
188 // If successful, disable automation
189 if( m_pExposureAuto != 0 )
190 m_pExposureAuto->setSwitch( enable );
191 }

219void CameraImagingSource::setExposure(double exposure) {
220 DShowLib::tIVCDPropertyItemsPtr pItems = handle.getAvailableVCDProperties();
221 if( pItems != 0 )
222 {
223 // Try to find the exposure item
224 DShowLib::tIVCDPropertyItemPtr pExposureItem = pItems->findItem( DShowLib::VCDID_Exposure );
225
226 DShowLib::tIVCDAbsoluteValuePropertyPtr m_pExposureValue;
227
228 DShowLib::tIVCDPropertyElementPtr pExposureValueElement = pExposureItem->findElement( DShowLib::VCDElement_Value );
229
230 // If a value element exists, try to acquire a range interface
231 if( pExposureValueElement != 0 )
232 {
233 pExposureValueElement->getInterfacePtr( m_pExposureValue );
234
235 double current_exposure = m_pExposureValue->getValue();
236 double exposure_d = exposure/100;
237
238 qDebug() << "current exposure: " << current_exposure;
239 qDebug() << "new exposure: " << exposure_d;
240
241 m_pExposureValue->setValue( exposure_d );
242 }
243 }

316void CameraImagingSource::setFrameRate(double setRate) {
317 bool was_live = false;
318
319 if (handle.isLive()) {
320 was_live = true;
321 }
322
323 stopCapture();
324 while (handle.isLive())
325 ;
326
327 if (setRate == 0){
328 setRate = handle.getCurrentMaxAvailableFPS();
329 }
330 if (!handle.setFPS(setRate)) {
331 debugMessage("Failed to change frame rate");
332 } else {
333 emit frameRateChanged(handle.getFPS());
334 }
335
336 qDebug() << "New frame rate" << handle.getFPS();
337
338 if (was_live) {
339 startCapture();
340 }
341
342 return;
343}
