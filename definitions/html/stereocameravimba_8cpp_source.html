8bool StereoCameraVimba::openCamera(){
9 if (isConnected()){
10 closeCamera();
11 }
12 int binning = stereoCameraSettings_.binning;
13 bool trigger;
14 if (stereoCameraSettings_.trigger == 1){
15 trigger = true;

17 trigger = false;
18 }
19 int fps = stereoCameraSettings_.fps;
20 double exposure = stereoCameraSettings_.exposure;
21 int gain = stereoCameraSettings_.gain;
22 bool autoExpose;
23 if (stereoCameraSettings_.autoExpose == 1){
24 autoExpose = true;

26 autoExpose = false;

28 bool autoGain;

30 autoGain = true;
31 } else {
32 autoGain = false;
33 }
34
35 this->connected = setupCameras(stereoCameraSerialInfo_,binning,trigger,fps);
36 enableAutoGain(autoGain);
37 enableAutoExposure(autoExpose);
38 setExposure(exposure);
39 setGain(gain);
40

53bool StereoCameraVimba::setupCameras(AbstractStereoCamera::StereoCameraSerialInfo CSI_cam_info,int iBinning, bool trigger, int iFps){
54 std::string camera_left_serial = CSI_cam_info.left_camera_serial;
55 std::string camera_right_serial = CSI_cam_info.right_camera_serial;
56
57 camera_left = new CameraVimba();
58 camera_right = new CameraVimba();
59
60 QThread* left_camera_thread = new QThread();
61 camera_left->assignThread(left_camera_thread);
62
63 QThread* right_camera_thread = new QThread();
64 camera_right->assignThread(right_camera_thread);
65
66 bool success = camera_left->initCamera(camera_left_serial, iBinning, trigger , iFps);
67 success &= camera_right->initCamera(camera_right_serial, iBinning, trigger, iFps);
68
69 if(success){
70 camera_left->getImageSize(image_width, image_height, image_bitdepth);
71 emit update_size(image_height, image_width, image_bitdepth);
72 }
73
74 connect(this, SIGNAL(start_capture()), camera_left, SLOT(startCapture()));
75 connect(this, SIGNAL(stop_capture()), camera_right, SLOT(startCapture()));
76 connect(this, SIGNAL(start_capture()), camera_left, SLOT(stopCapture()));
77 connect(this, SIGNAL(stop_capture()), camera_right, SLOT(stopCapture()));
78
79 return success;
80}

82bool StereoCameraVimba::captureSingle(){
83
84 camera_left->stopCapture();
85 camera_right->stopCapture();
86
87 auto future_left = QtConcurrent::run(camera_left, &CameraVimba::capture);
88 auto future_right = QtConcurrent::run(camera_right, &CameraVimba::capture);
89
90 future_left.waitForFinished();
91 future_right.waitForFinished();
92
93 bool res = false;
94
95 if(future_left.result() && future_right.result()){
96 camera_left->getImage(left_raw);
97 camera_right->getImage(right_raw);
98 emit captured_success();
99 res = true;
100 }else{
101 emit captured_fail();
102 qDebug() << "Capture failed";
103 res = false;
104 }
105
106 emit captured();
107 return res;
108}

155

168 camera_right->stopCapture();

170 }

174std::vector<AbstractStereoCamera::StereoCameraSerialInfo> StereoCameraVimba::listSystems(){
175
176 using namespace AVT::VmbAPI;
177
178 std::vector<AbstractStereoCamera::StereoCameraSerialInfo> known_serial_infos = loadSerials(CAMERA_TYPE_VIMBA);
179 std::vector<AbstractStereoCamera::StereoCameraSerialInfo> connected_serial_infos;
180
181 // NB you need to explicitly state the left hand type, otherwise
182 // you'll get weird private constructor errors - don't use auto
183
184 VimbaSystem& system = VimbaSystem::GetInstance();
185 CameraPtrVector all_cameras;
186
187 auto err = system.GetCameras( all_cameras ); // Fetch all cameras known to Vimba
188 if( err != VmbErrorSuccess ){
189 qDebug() << "Could not list cameras. Error code: " << err << "\n";
190 return connected_serial_infos;
191 }
192
193 qDebug() << "Cameras found: " << all_cameras.size() <<"\n\n";
194
195 if(all_cameras.size() == 0){
196 return connected_serial_infos;
197 }
198
199 std::vector<std::string> connected_camera_names;
200 std::vector<std::string> connected_serials;
201 //TODO add generic way to recognise i3dr cameras whilst still being
202 //able to make sure the correct right and left cameras are selected together
203 for (size_t i = 0; i < all_cameras.size(); ++i)
204 {
205 std::string device_serial;
206 VmbInterfaceType interface_type;
207 all_cameras[i]->GetSerialNumber(device_serial);
208 //TODO get device_name
209 all_cameras[i]->GetInterfaceType(interface_type);
210 if (interface_type == VmbInterfaceUsb){
211 connected_serials.push_back(device_serial);
212 //connected_camera_names.push_back(device_name);
213 } else {
214 qDebug() << "Unsupported vimba class: " << interface_type;
215 }
216 }
217
218 for (auto& known_serial_info : known_serial_infos) {
219 bool left_found = false;
220 bool right_found = false;
221 std::string left_serial;
222 std::string right_serial;
223 //find left
224 for (auto& connected_serial : connected_serials)
225 {
226 left_serial = connected_serial;
227 if (left_serial == known_serial_info.left_camera_serial){
228 left_found = true;
229 break;
230 }
231 }
232 if (left_found){
233 //find right
234 for (auto& connected_serial : connected_serials)
235 {
236 right_serial = connected_serial;
237 if (right_serial == known_serial_info.right_camera_serial){
238 right_found = true;
239 break;
240 }
241 }
242 }
243 if (left_found && right_found){ //only add if both cameras found
244 connected_serial_infos.push_back(known_serial_info);
245 }
246 }
247
248 return connected_serial_infos;
249}
