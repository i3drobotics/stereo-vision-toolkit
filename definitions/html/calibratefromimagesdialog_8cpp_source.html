87void CalibrateFromImagesDialog::setLeftImages() {
88 left_paths_checked = true;
89 QModelIndex parentIndex = left_file_model->index(left_path);
90 int numRows = left_file_model->rowCount(parentIndex);
91 // left_image_list = QList<cv::Mat>();
92 left_image_path_list = QList<std::string>();
93
94 for (int row = 0; row < numRows; ++row) {
95 QModelIndex childIndex = left_file_model->index(row, 0, parentIndex);
96 QString path = left_file_model->data(childIndex).toString();
97 QString fname = left_path + "/" + path;
98 // cv::Mat im = cv::imread(fname.toStdString(), cv::IMREAD_GRAYSCALE);
99 // left_image_list.append(im);
100 left_image_path_list.append(fname.toStdString());
101 }
102
103 if (left_paths_checked && right_paths_checked)
104 checkImageCount();
105}

127void CalibrateFromImagesDialog::findImages() {
128 left_paths_checked = false;
129 right_paths_checked = false;
130 if (left_path != "") {
131 QStringList leftFilters;
132
133 if (left_mask == "") {
134 leftFilters << "*.png"
135 << "*.tif"
136 << "*.jpg"
137 << "*.jpeg";
138 } else {
139 leftFilters << left_mask;
140 }
141
142 left_file_model->setNameFilters(leftFilters);
143 left_file_model->setRootPath(left_path);
144 left_file_model->setNameFilterDisables(false);
145
146 ui->leftImageTable->setModel(left_file_model);
147 ui->leftImageTable->setRootIndex(left_file_model->index(left_path));
148
149 } else {
150 ui->leftImageTable->clearSpans();
151 }
152
153 if (right_path != "") {
154 QStringList rightFilters;
155
156 if (right_mask == "") {
157 rightFilters << "*.png"
158 << "*.tif"
159 << "*.jpg"
160 << "*.jpeg";
161 } else {
162 rightFilters << right_mask;
163 }
164
165 right_file_model->setNameFilters(rightFilters);
166 right_file_model->setRootPath(right_path);
167 right_file_model->setNameFilterDisables(false);
168
169 ui->rightImageTable->setModel(right_file_model);
170 ui->rightImageTable->setRootIndex(right_file_model->index(right_path));
171 } else {
172 ui->rightImageTable->clearSpans();
173 }

176void CalibrateFromImagesDialog::checkImageCount(void){
177 // Only allow user to continue if we've got enough cal images
178 // int n_left_images = left_image_list.size();
179 // int n_right_images = right_image_list.size();
180 int n_left_images = left_image_path_list.size();
181 int n_right_images = right_image_path_list.size();
182
183 if(n_left_images >= 5 && n_right_images >= 5 && n_left_images == n_right_images){
184 ui->okPushButton->setEnabled(true);
185 }else{
186 ui->okPushButton->setEnabled(false);
187 QMessageBox alert;
188 alert.setText("Minimum of 5 images required for stereo calibration. Recommend 15-20, see help section in the toolbar for details on calibration.");
189 alert.exec();
190 }
191}
