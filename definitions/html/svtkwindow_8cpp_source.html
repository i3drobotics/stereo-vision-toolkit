465void SVTKWindow::enableSharedMemory(bool enable){
466 if (!stereo_cam){
467 enable = false;
468 }
469 if (enable){
470 std::string mapname = ui->txtSharedMemMap->text().toStdString();
471 std::string mutexname = ui->txtSharedMemMutex->text().toStdString();
472 int shared_mem_downsample_factor = ui->spinBoxSharedMemDownsample->value();
473 int camera_downsample_factor = ui->spinBoxImageDownsample->value();
474 double downsample_rate = (1.0/(double)shared_mem_downsample_factor) * (1.0/(double)camera_downsample_factor);
475 int img_height = round(stereo_cam->getHeight() * downsample_rate);
476 int img_width = round(stereo_cam->getWidth() * downsample_rate);
477 if (ui->comboBoxSharedMemSource->currentText() == "RGBD"){
478 connect(stereo_cam, SIGNAL(matched()), this, SLOT(updateSharedMemory()));
479 // clear shared memory if open
480 if (sharedMemoryInst->isOpen()){
481 sharedMemoryInst->close();
482 }
483 if (ui->checkBoxSharedMem16bit->isChecked()){
484 // create shared memory file to store 4 channel rgbd image with element size 4*unsigned short size
485 sharedMemoryInst->open(img_height, img_width, 4, 4*sizeof(unsigned short), CV_16UC4, mapname, mutexname);
486 } else {
487 // create shared memory file to store 4 channel rgbd image with element size 4*float size
488 sharedMemoryInst->open(img_height, img_width, 4, 4*sizeof(float), CV_32FC4, mapname, mutexname);
489 }
490 } else {
491 connect(stereo_cam, SIGNAL(stereopair_processed()), this, SLOT(updateSharedMemory()));
492 // clear shared memory if open
493 if (sharedMemoryInst->isOpen()){
494 sharedMemoryInst->close();
495 }
496 // create shared memory file to store 3 channel rgb image with element size 3*uchar size
497 sharedMemoryInst->open(img_height, img_width, 3, 3*sizeof(unsigned char), CV_8UC3, mapname, mutexname);
498 }
499 } else {
500 disconnect(stereo_cam, SIGNAL(stereopair_processed()), this, SLOT(updateSharedMemory()));
501 disconnect(stereo_cam, SIGNAL(matched()), this, SLOT(updateSharedMemory()));
502 //stop shared memory server
503 sharedMemoryInst->close();
504 }
505 shared_memory_enabled = enable;
506 ui->txtSharedMemMap->setEnabled(!enable);
507 ui->txtSharedMemMutex->setEnabled(!enable);
508 ui->comboBoxSharedMemSource->setEnabled(!enable);
509 ui->checkBoxSharedMemUseRectified->setEnabled(!enable);
510 ui->checkBoxSharedMem16bit->setEnabled(!enable);
511 ui->spinBoxSharedMemDownsample->setEnabled(!enable);
512}

757void SVTKWindow::updateDetection(){
758
759 if(!object_detector) return;
760
761 // Protect callback against high frame rates (faster than detector)
762 if(object_detector->isRunning() || detecting){
763 return;
764 }
765
766 this->detecting = true;
767
768 // Select image source
769 if(ui->imageSourceComboBox->currentText() == "Left"){
770 stereo_cam->getLeftImage(image_detection);
771 }else{
772 stereo_cam->getRightImage(image_detection);
773 }
774
775 double scale_factor_x = 1.0;
776 double scale_factor_y = 1.0;
777
778 if(detection_enabled){
779
780 if(image_detection.empty()){
781 qDebug() << "Empty image passed to detector";
782 return;
783 }
784
785 // Check if we have a network loaded
786 if(!object_detector->isReady())
787 return;
788
789 // We'll later resize the bounding boxes to the input image size
790 scale_factor_x = static_cast<double>(object_detector->getInputWidth())/image_detection.cols;
791 scale_factor_y = static_cast<double>(object_detector->getInputHeight())/image_detection.rows;
792
793 // (Down)scale input for DNN first layer
794 cv::resize(image_detection,
795 image_detection_rescale,
796 cv::Size(),
797 scale_factor_x,
798 scale_factor_y);
799
800
801 // Inference
802 auto results = object_detector->infer(image_detection_rescale);
803
804 // Update UI
805 ui->latencyLabel->setText(QString("%1 ms").arg(object_detector->getProcessingTime()));
806 ui->numberObjectsLabel->setText(QString("%1").arg(results.size()));
807
808 // Draw bounding boxes, but we need a 4-channel image with alpha
809 if(image_detection.channels() == 1){
810 cv::cvtColor(image_detection, image_detection, cv::COLOR_GRAY2RGBA);
811 }else if(image_detection.channels() == 3){
812 cv::cvtColor(image_detection, image_detection, cv::COLOR_BGR2RGBA); //Assumes image from camera is BGR
813 }
814 drawBoundingBoxes(image_detection, results, 1./scale_factor_x, 1./scale_factor_y);
815
816 cv::cvtColor(image_detection, image_detection, cv::COLOR_RGBA2BGRA);
817
818 object_detection_display->updateView(image_detection);
819 QCoreApplication::processEvents();
820 }
821
822 this->detecting = false;
823}

825void SVTKWindow::drawBoundingBoxes(cv::Mat image, std::vector<BoundingBox> bboxes, double scale_x=1.0, double scale_y=1.0){
826
827 for(auto &bbox : bboxes){
828
829 if(!class_visible_map[bbox.classname])
830 continue;
831
832 cv::Point top_left(static_cast<int>(scale_x*bbox.rect.topLeft().x()),
833 static_cast<int>(scale_y*bbox.rect.topLeft().y()));
834 cv::Point bottom_right(static_cast<int>(scale_x*bbox.rect.bottomRight().x()),
835 static_cast<int>(scale_y*bbox.rect.bottomRight().y()));
836 int thickness = 2;
837
838 QColor box_colour = class_colour_map[bbox.classname];
839 int r=255, g=0, b=0, a=255; // default to red
840 box_colour.getRgb(&r, &g, &b, &a);
841
842 cv::Scalar font_colour(r, g, b, a);
843
844 //Get the label for the class name and its confidence
845 std::string label = cv::format("%s %.2f", bbox.classname.toStdString().c_str(), bbox.confidence);
846
847 float fontScale = (float)image.size().height / 1200;
848 float fontThickness = fontScale*2;
849
850 //Display the label at the top of the bounding box
851 int baseLine;
852 cv::Size labelSize = getTextSize(label, cv::FONT_HERSHEY_COMPLEX, fontScale, 1, &baseLine);
853
854 cv::Point text_pos = top_left;
855 text_pos.y = text_pos.y - (labelSize.height / 2);
856 cv::putText(image, label, text_pos, cv::FONT_HERSHEY_COMPLEX, fontScale, font_colour, fontThickness);
857
858 if(class_filled_map[bbox.classname]){
859 cv::Scalar font_colour(r, g, b);
860 thickness = -1;
861
862 cv::Rect bbox_roi(top_left, bottom_right);
863 cv::Mat subimage = image(bbox_roi);
864 cv::Mat subimage_overlay = subimage.clone();
865 subimage_overlay = cv::Scalar(r, g, b);
866
867 // Overlay box
868 double gamma = 0.0;
869 double alpha = static_cast<double>(bounding_box_alpha) / 255.0;
870 cv::addWeighted(subimage_overlay, alpha, subimage, 1-alpha, gamma, subimage);
871
872 //image(roi) = subimage;
873 }else{
874 cv::rectangle(image, top_left, bottom_right, font_colour, thickness);
875 }
876 }
877}

1325void SVTKWindow::updateSharedMemory(){
1326 if (shared_memory_enabled){
1327 //qDebug() << "Getting source image...";
1328 bool useRectified = ui->checkBoxSharedMemUseRectified->isChecked();
1329 cv::Mat left_t, right_t;
1330 cv::Mat wImage, Q;
1331 double downsample_rate = 1.0f/(double)ui->spinBoxSharedMemDownsample->value();
1332 if (ui->comboBoxSharedMemSource->currentText() == "Stereo"){
1333 if (useRectified){
1334 stereo_cam->getLeftImage(left_t);
1335 stereo_cam->getRightImage(right_t);
1336 } else {
1337 stereo_cam->getLeftRawImage(left_t);
1338 stereo_cam->getRightRawImage(right_t);
1339 }
1340 // convert image to rgb
1341 if (left_t.type() == CV_8UC1){
1342 cvtColor(left_t,left_t,COLOR_GRAY2RGB);
1343 } else if (left_t.type() == CV_8UC3){
1344 cvtColor(left_t,left_t,COLOR_BGR2RGB);
1345 } else {
1346 std::cerr << "Invalid image type for shared memory" << std::endl;
1347 }
1348 if (right_t.type() == CV_8UC1){
1349 cvtColor(right_t,right_t,COLOR_GRAY2RGB);
1350 } else if (right_t.type() == CV_8UC3){
1351 cvtColor(right_t,right_t,COLOR_BGR2RGB);
1352 } else {
1353 std::cerr << "Invalid image type for shared memory" << std::endl;
1354 }
1355 cv::hconcat(left_t, right_t, image_stream);
1356 cv::resize(image_stream, image_stream, cv::Size(), downsample_rate, downsample_rate);
1357 } else if (ui->comboBoxSharedMemSource->currentText() == "RGBD"){
1358 if (stereo_cam->isMatching()){
1359 cv::Mat color, Q, depth, depth_z, depth_split[3];
1360 stereo_cam->getLeftMatchImage(color);
1361 stereo_cam->getDepth(depth);
1362 stereo_cam->getQ(Q);
1363 // extract Z only channel from depth (xyz) image
1364 cv::split(depth, depth_split);
1365 depth_z = depth_split[2];
1366 // get horizontal fov from Q matrix
1367 float hfov = CVSupport::getHFOVFromQ(Q);
1368 // embed horizontal fov in top left pixel of z only depth image to simplify reconstruction
1369 depth_z.at<float>(0,0) = (float)hfov;
1370 if (!depth_z.empty() && !color.empty()){
1371 // convert image to rgb
1372 if (color.type() == CV_8UC1){
1373 cvtColor(color,color,CV_GRAY2RGB);
1374 } else if (color.type() == CV_8UC3){
1375 cvtColor(color,color,CV_BGR2RGB);
1376 } else {
1377 std::cerr << "Invalid image type for shared memory" << std::endl;
1378 }
1379 // Downsample images based on parameters
1380 cv::resize(depth_z, depth_z, cv::Size(), downsample_rate, downsample_rate, cv::INTER_NEAREST_EXACT);
1381 cv::resize(color, color, cv::Size(), downsample_rate, downsample_rate);
1382 // Create RGBD image from color image and z only depth image
1383 if (ui->checkBoxSharedMem16bit->isChecked()){
1384 image_stream = CVSupport::createRGBD16(color,depth_z,6553.0,false);
1385 } else {
1386 image_stream = CVSupport::createRGBD32(color,depth_z);
1387 }
1388 } else {
1389 image_stream = cv::Mat();
1390 std::cerr << "Depth image or camera image is empty." << std::endl;
1391 }
1392 } else {
1393 std::cerr << "Missing disparity for sending rgbd in shared memory." << std::endl;
1394 image_stream = cv::Mat();
1395 }
1396 } else {
1397 std::cerr << "Invalid shared memory source: " << ui->comboBoxSharedMemSource->currentText().toStdString();
1398 }
1399
1400 if(image_stream.empty()){
1401 qDebug() << "Empty image passed to shared memory";
1402 return;
1403 }
1404
1405 sharedMemoryInst->write_threaded(image_stream);
1406 }
