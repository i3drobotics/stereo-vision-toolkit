15void CStereoCameraBaslerImageEventHandler::OnImageGrabbed( Pylon::CInstantCamera& camera, const Pylon::CGrabResultPtr& ptrGrabResult)
16{
17 const auto now = std::chrono::system_clock::now();
18 int frame_timestamp_milli = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
19 // left camera
20 if (camera.GetDeviceInfo().GetSerialNumber() == this->left_serial.c_str()){
21 bool res = PylonSupport::grabImage2mat(ptrGrabResult,this->formatConverter,this->live_left_image);
22 if (res){
23 timestamp_left = frame_timestamp_milli;
24 }
25 }
26 // right camera
27 if (camera.GetDeviceInfo().GetSerialNumber() == this->right_serial.c_str()){
28 bool res = PylonSupport::grabImage2mat(ptrGrabResult,this->formatConverter,this->live_right_image);
29 if (res){
30 timestamp_right = frame_timestamp_milli;
31 }
32 }
33 mtx.lock();
34 int lr_diff = abs(timestamp_left - timestamp_right);
35 if (lr_diff <= 10){
36 newFrames = true;
37 this->left_image = this->live_left_image.clone();
38 this->right_image = this->live_right_image.clone();
39 } else {
40 newFrames = false;
41 }
42 mtx.unlock();
43}

45bool CStereoCameraBaslerImageEventHandler::getImagePair(cv::Mat &left, cv::Mat &right, int timeout=1000){
46 int time_count = 0;
47 bool timed_out = true;
48 while(time_count < timeout){
49 mtx.lock();
50 if (newFrames){
51 left = this->left_image.clone();
52 right = this->right_image.clone();
53 newFrames = false;
54 timed_out = false;
55 mtx.unlock();
56 break;
57 }
58 mtx.unlock();
59 std::this_thread::sleep_for(std::chrono::milliseconds(1));
60 time_count += 1;
61 }
62 return !timed_out;
63}

72bool StereoCameraBasler::openCamera(){
73 camControl = new ArduinoCommsCameraControl();
74 std::vector<QSerialPortInfo> serial_device_list = camControl->getSerialDevices();
75 if (serial_device_list.size() <= 0){
76 qDebug() << "Failed to find serial device for hardware control.";
77 //return false;
78 } else {
79 has_trigger_fps_control = camControl->open(serial_device_list.at(0),115200);
80 }
81
82 int binning = stereoCameraSettings_.binning;
83 bool trigger;
84 if (stereoCameraSettings_.trigger == 1){
85 trigger = true;
86 } else {
87 trigger = false;
88 }
89 hardware_triggered = trigger;
90 int fps = stereoCameraSettings_.fps;;
91 double exposure = stereoCameraSettings_.exposure;
92 int gain = stereoCameraSettings_.gain;
93 int packet_size = stereoCameraSettings_.packetSize;
94 int packet_delay = stereoCameraSettings_.packetDelay;
95 bool autoExpose;
96 if (stereoCameraSettings_.autoExpose == 1){
97 autoExpose = true;
98 } else {
99 autoExpose = false;
100 }
101 bool autoGain;
102 if (stereoCameraSettings_.autoGain == 1){
103 autoGain = true;
104 } else {
105 autoGain = false;
106 }
107
108 try
109 {
110
111 // Create an instant camera object with the camera device found first.
112 Pylon::CTlFactory& tlFactory = Pylon::CTlFactory::GetInstance();
113
114 Pylon::DeviceInfoList_t devices;
115 tlFactory.EnumerateDevices(devices);
116
117 this->cameras = new Pylon::CInstantCameraArray(2);
118 Pylon::CInstantCameraArray all_cameras(devices.size());
119
120 std::string camera_left_serial = stereoCameraSerialInfo_.left_camera_serial;
121 std::string camera_right_serial = stereoCameraSerialInfo_.right_camera_serial;
122
123 bool cameraLeftFind = false;
124 for (size_t i = 0; i < all_cameras.GetSize(); ++i)
125 {
126 all_cameras[i].Attach(tlFactory.CreateDevice(devices[i]));
127
128 if (all_cameras[i].GetDeviceInfo().GetSerialNumber() == camera_left_serial.c_str())
129 {
130 cameras->operator[](0).Attach(tlFactory.CreateDevice(devices[i]));
131 cameraLeftFind = true;
132 break;
133 }
134 }
135
136 if (!cameraLeftFind){
137 std::cerr << "Failed to find left camera with serial: " << camera_left_serial << std::endl;
138 return false;
139 }
140
141 bool cameraRightFind = false;
142 for (size_t i = 0; i < all_cameras.GetSize(); ++i)
143 {
144 all_cameras[i].Attach(tlFactory.CreateDevice(devices[i]));
145
146 if (all_cameras[i].GetDeviceInfo().GetSerialNumber() == camera_right_serial.c_str())
147 {
148 cameras->operator[](1).Attach(tlFactory.CreateDevice(devices[i]));
149 cameraRightFind = true;
150 break;
151 }
152 }
153
154 if (!cameraRightFind){
155 std::cerr << "Failed to find right camera with serial: " << camera_right_serial << std::endl;
156 return false;
157 }
158
159 // Set device link throughput to fix data collision
160 enableDeviceLinkThroughputLimit(true);
161 setDeviceLinkThroughput(100000000);
162
163 //check open
164 bool connected = true;
165 try{
166 for (size_t i = 0; i < cameras->GetSize(); ++i)
167 {
168 cameras->operator[](i).Open();
169 connected &= cameras->operator[](i).IsOpen();
170 }
171 }
172 catch (const Pylon::GenericException &e)
173 {
174 // Error handling.
175 std::cerr << "An exception occurred whilst opening camera." << std::endl
176 << e.GetDescription() << std::endl;
177 connected = false;
178 }
179
180 if (!connected){
181 return connected;
182 }
183
184 // Flip camera image if required (due to mounting orientation)
185 Pylon::CBooleanParameter(cameras->operator[](0).GetNodeMap(), "ReverseX").SetValue(stereoCameraSettings_.flip_left_x);
186 Pylon::CBooleanParameter(cameras->operator[](0).GetNodeMap(), "ReverseY").SetValue(stereoCameraSettings_.flip_left_y);
187 Pylon::CBooleanParameter(cameras->operator[](1).GetNodeMap(), "ReverseX").SetValue(stereoCameraSettings_.flip_right_x);
188 Pylon::CBooleanParameter(cameras->operator[](1).GetNodeMap(), "ReverseY").SetValue(stereoCameraSettings_.flip_right_y);
189
190 getImageSize(cameras->operator[](0),image_width,image_height,image_bitdepth);
191 emit update_size(image_width, image_height, image_bitdepth);
192
193 formatConverter = new Pylon::CImageFormatConverter();
194
195 //formatConverter->OutputPixelFormat = Pylon::PixelType_Mono8;
196 formatConverter->OutputPixelFormat = Pylon::PixelType_BGR8packed;
197 formatConverter->OutputBitAlignment = Pylon::OutputBitAlignment_MsbAligned;
198
199 imageEventHandler = new CStereoCameraBaslerImageEventHandler(camera_left_serial,camera_right_serial,formatConverter);
200
201 for (size_t i = 0; i < cameras->GetSize(); ++i)
202 {
203 //cameras->operator[](i).MaxNumBuffer = 1;
204 //cameras->operator[](i).OutputQueueSize = 1;
205 //cameras->operator[](i).ClearBufferModeEnable();
206 cameras->operator[](i).RegisterImageEventHandler( imageEventHandler, Pylon::RegistrationMode_Append, Pylon::Cleanup_Delete);
207 }
208 }
209 catch (const Pylon::GenericException &e)
210 {
211 // Error handling.
212 std::cerr << "An exception occurred whilst setting up cameras." << std::endl
213 << e.GetDescription() << std::endl;
214 connected = false;
215 return connected;
216 }
217
218 // Set device parameters
219 setBinning(binning);
220 enableTrigger(trigger);
221 setPacketSize(packet_size);
222 setFPS(fps);
223
224 enableAutoGain(autoGain);
225 enableAutoExposure(autoExpose);
226 setExposure(exposure);
227 setGain(gain);
228 setPacketDelay(packet_delay);
229
230 //lineStatusTimerId = startTimer(lineStatusTimerDelay);
231
232 connected = true;
233 return connected;
234}

237 std::vector<AbstractStereoCamera::StereoCameraSerialInfo> known_serial_infos_gige = loadSerials(AbstractStereoCamera::CAMERA_TYPE_BASLER_GIGE);

245 // Initialise Basler Pylon

249 //Pylon::CTlFactory& tlFactory = Pylon::CTlFactory::GetInstance();
250
251 Pylon::DeviceInfoList_t devices;
252 tlFactory->EnumerateDevices(devices);
253
254 qDebug() << "Time to initalise pylon: " << task_timer.elapsed();

259 std::vector<std::string> connected_camera_names;
260 std::vector<std::string> connected_serials;
261 //TODO add generic way to recognise i3dr cameras whilst still being
262 //able to make sure the correct right and left cameras are selected together
263 for (size_t i = 0; i < devices.size(); ++i)
264 {
265 std::string device_class = std::string(devices[i].GetDeviceClass());
266 std::string device_name = std::string(devices[i].GetUserDefinedName());
267 std::string device_serial = std::string(devices[i].GetSerialNumber());
268 if (device_class == DEVICE_CLASS_GIGE || device_class == DEVICE_CLASS_USB){
269 connected_serials.push_back(device_serial);
270 connected_camera_names.push_back(device_name);
271 } else {
272 qDebug() << "Unsupported basler class: " << device_class.c_str();
273 }
274 }
275
276 for (auto& known_serial_info : known_serial_infos) {
277 bool left_found = false;
278 bool right_found = false;
279 std::string left_serial;
280 std::string right_serial;
281 //find left
282 for (auto& connected_serial : connected_serials)
283 {
284 left_serial = connected_serial;
285 if (left_serial == known_serial_info.left_camera_serial){
286 left_found = true;
287 break;
288 }
289 }
290 if (left_found){
291 //find right
292 for (auto& connected_serial : connected_serials)
293 {
294 right_serial = connected_serial;
295 if (right_serial == known_serial_info.right_camera_serial){
296 right_found = true;
297 break;
298 }
299 }
300 }
301 if (left_found && right_found){ //only add if both cameras found
302 connected_serial_infos.push_back(known_serial_info);
303 }
304 }
305
306 //Pylon::PylonTerminate();
307 return connected_serial_infos;

514bool StereoCameraBasler::setFPS(int val){
515 if (hardware_triggered){
516 qDebug() << "Setting hardware fps..." << val;
517 if (!camControl->isConnected()){
518 std::vector<QSerialPortInfo> serial_device_list = camControl->getSerialDevices();
519 if (serial_device_list.size() <= 0){
520 qDebug() << "Failed to find serial device for hardware control.";
521 } else {
522 camControl->open(serial_device_list.at(0),115200);
523 }
524 if (camControl->isConnected()){
525 camControl->updateFPS(val);
526 } else {
527 qDebug() << "Failed to open connection to camera control device";
528 // TODO remove this when camcontrol consistently working
529 frame_rate = val;
530 return false;
531 }
532 } else {
533 camControl->updateFPS(val);
534 frame_rate = val;
535 return true;
536 }
537 } else {
538 try
539 {
540 qDebug() << "Setting software fps..." << val;
541 float fps_f = (float)val;
542 for (size_t i = 0; i < cameras->GetSize(); ++i)
543 {
544 cameras->operator[](i).Open();
545 if (stereoCameraSettings_.isGige){
546 Pylon::CFloatParameter(cameras->operator[](i).GetNodeMap(), "AcquisitionFrameRateAbs").SetValue(fps_f);
547 } else {
548 Pylon::CFloatParameter(cameras->operator[](i).GetNodeMap(), "AcquisitionFrameRate").SetValue(fps_f);
549 }
550 }
551 frame_rate = val;
552 return true;
553 }
554 catch (const Pylon::GenericException &e)
555 {
556 // Error handling.
557 std::cerr << "An exception occurred." << std::endl
558 << e.GetDescription() << std::endl;
559 return false;
560 }
561 }
562 return false;
563}
