26bool AbstractArduinoComs::open(QSerialPortInfo serial_port_info,int baudrate){
27 m_baudrate = baudrate;
28 m_serialPortInfo = serial_port_info;
29 m_timer->setSingleShot(true);
30 m_serialPort->setPort(serial_port_info);
31 m_serialPort->setBaudRate(baudrate);
32 connected = m_serialPort->open(QIODevice::ReadWrite);
33
34 if (connected){
35 connect(m_serialPort, &QSerialPort::bytesWritten,
36 this, &AbstractArduinoComs::handleBytesWritten);
37 connect(m_serialPort, &QSerialPort::errorOccurred,
38 this, &AbstractArduinoComs::handleError);
39 connect(m_timer, &QTimer::timeout, this, &AbstractArduinoComs::handleTimeout);
40 }
41
42 return connected;
43}

45void AbstractArduinoComs::handleBytesWritten(qint64 bytes)
46{
47 m_bytesWritten += bytes;
48 if (m_bytesWritten == m_writeData.size()) {
49 m_bytesWritten = 0;

51 .arg(m_serialPort->portName()) << "\n";

74void AbstractArduinoComs::write(const QString &writeData)
75{
76 m_writeData = writeData.toUtf8();
77
78 const qint64 bytesWritten = m_serialPort->write(m_writeData);
79
80 if (bytesWritten == -1) {
81 m_standardOutput << QObject::tr("Failed to write the data to port %1, error: %2")
82 .arg(m_serialPort->portName(),
83 m_serialPort->errorString())
84 << "\n";
85 } else if (bytesWritten != m_writeData.size()) {
86 m_standardOutput << QObject::tr("Failed to write all the data to port %1, error: %2")
87 .arg(m_serialPort->portName(),
88 m_serialPort->errorString())
89 << "\n";
90 }
91
92 m_timer->start(5000);
93}
