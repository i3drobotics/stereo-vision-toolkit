56 // NOTE: This does not cover all cases - it should be easy to add new ones as required.
57 inline QImage cvMatToQImage( const cv::Mat &inMat )
58 {
59 switch ( inMat.type() )
60 {
61 // 8-bit, 4 channel
62 case CV_8UC4:
63 {
64 QImage image( inMat.data,
65 inMat.cols, inMat.rows,
66 static_cast<int>(inMat.step),
67 QImage::Format_ARGB32 );
68
69 return image;
70 }
71
72 // 8-bit, 3 channel
73 case CV_8UC3:
74 {
75 QImage image( inMat.data,
76 inMat.cols, inMat.rows,
77 static_cast<int>(inMat.step),
78 QImage::Format_RGB888 );
79
80 return image.rgbSwapped();
81 }
82
83 // 8-bit, 1 channel
84 case CV_8UC1:
85 {
86 #if QT_VERSION >= QT_VERSION_CHECK(5, 5, 0)
87 QImage image( inMat.data,
88 inMat.cols, inMat.rows,
89 static_cast<int>(inMat.step),
90 QImage::Format_Grayscale8 );
91 #else
92 static QVector<QRgb> sColorTable;
93
94 // only create our color table the first time
95 if ( sColorTable.isEmpty() )
96 {
97 sColorTable.resize( 256 );
98
99 for ( int i = 0; i < 256; ++i )
100 {
101 sColorTable[i] = qRgb( i, i, i );
102 }
103 }
104
105 QImage image( inMat.data,
106 inMat.cols, inMat.rows,
107 static_cast<int>(inMat.step),
108 QImage::Format_Indexed8 );
109
110 image.setColorTable( sColorTable );
111 #endif
112
113 return image;
114 }
115
116 default:
117 qWarning() << "ASM::cvMatToQImage() - cv::Mat image type not handled in switch:" << inMat.type();
118 break;
119 }
120
121 return QImage();
122 }

129 // If inImage exists for the lifetime of the resulting cv::Mat, pass false to inCloneImageData to share inImage's
130 // data with the cv::Mat directly
131 // NOTE: Format_RGB888 is an exception since we need to use a local QImage and thus must clone the data regardless
132 // NOTE: This does not cover all cases - it should be easy to add new ones as required.
133 inline cv::Mat QImageToCvMat( const QImage &inImage, bool inCloneImageData = true )
134 {
135 switch ( inImage.format() )
136 {
137 // 8-bit, 4 channel
138 case QImage::Format_ARGB32:
139 case QImage::Format_ARGB32_Premultiplied:
140 {
141 cv::Mat mat( inImage.height(), inImage.width(),
142 CV_8UC4,
143 const_cast<uchar*>(inImage.bits()),
144 static_cast<size_t>(inImage.bytesPerLine())
145 );
146
147 return (inCloneImageData ? mat.clone() : mat);
148 }
149
150 // 8-bit, 3 channel
151 case QImage::Format_RGB32:
152 {
153 if ( !inCloneImageData )
154 {
155 qWarning() << "ASM::QImageToCvMat() - Conversion requires cloning so we don't modify the original QImage data";
156 }
157
158 cv::Mat mat( inImage.height(), inImage.width(),
159 CV_8UC4,
160 const_cast<uchar*>(inImage.bits()),
161 static_cast<size_t>(inImage.bytesPerLine())
162 );
163
164 cv::Mat matNoAlpha;
165
166 cv::cvtColor( mat, matNoAlpha, cv::COLOR_BGRA2BGR ); // drop the all-white alpha channel
167
168 return matNoAlpha;
169 }
170
171 // 8-bit, 3 channel
172 case QImage::Format_RGB888:
173 {
174 if ( !inCloneImageData )
175 {
176 qWarning() << "ASM::QImageToCvMat() - Conversion requires cloning so we don't modify the original QImage data";
177 }
178
179 QImage swapped = inImage.rgbSwapped();
180
181 return cv::Mat( swapped.height(), swapped.width(),
182 CV_8UC3,
183 const_cast<uchar*>(swapped.bits()),
184 static_cast<size_t>(swapped.bytesPerLine())
185 ).clone();
186 }
187
188 // 8-bit, 1 channel
189 case QImage::Format_Indexed8:
190 {
191 cv::Mat mat( inImage.height(), inImage.width(),
192 CV_8UC1,
193 const_cast<uchar*>(inImage.bits()),
194 static_cast<size_t>(inImage.bytesPerLine())
195 );
196
197 return (inCloneImageData ? mat.clone() : mat);
198 }
199
200 default:
201 qWarning() << "ASM::QImageToCvMat() - QImage format not handled in switch:" << inImage.format();
202 break;
203 }
204
205 return cv::Mat();
206 }

208 // If inPixmap exists for the lifetime of the resulting cv::Mat, pass false to inCloneImageData to share inPixmap's data
209 // with the cv::Mat directly
210 // NOTE: Format_RGB888 is an exception since we need to use a local QImage and thus must clone the data regardless
211 inline cv::Mat QPixmapToCvMat( const QPixmap &inPixmap, bool inCloneImageData = true )
212 {
213 return QImageToCvMat( inPixmap.toImage(), inCloneImageData );
