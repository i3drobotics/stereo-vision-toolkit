49void DetectorOpenCV::setTarget(int target){
50 preferable_target = target;
51
52 if(preferable_target == cv::dnn::DNN_TARGET_OPENCL){
53 // Check for GPU
54 cv::ocl::Context context;
55
56 if(!cv::ocl::haveOpenCL()){
57 std::cout << "OpenCL is not available. Falling back to CPU" << std::endl;
58 preferable_target = cv::dnn::DNN_TARGET_CPU;
59 }
60
61 // Attempt to use a GPU
62 if(context.create(cv::ocl::Device::TYPE_GPU)){
63 std::cout << "Found OpenCL capable GPU - we're going to use it!" << std::endl;
64 cv::ocl::Device(context.device(1));
65 }
66 }
67#ifdef WITH_CUDA
68 else if(preferable_target == cv::dnn::DNN_TARGET_CUDA || preferable_target == cv::dnn::DNN_TARGET_CUDA_FP16){
69 try
70 {
71 // Check for GPU
72 if (cv::cuda::getCudaEnabledDeviceCount() <= 0) {
73 qDebug() << "CUDA is not available. Falling back to CPU";
74 preferable_target = cv::dnn::DNN_TARGET_CPU;
75 }else{
76 qDebug() << "NVIDIA GPU detected.";
77 }
78 }
79 catch( cv::Exception& e )
80 {
81 const char* err_msg = e.what();
82 std::cout << "exception caught: " << err_msg << std::endl;
83 qDebug() << "CUDA is not available. Falling back to CPU";
84 preferable_target = cv::dnn::DNN_TARGET_CPU;
85 }
86 }
87#endif

186void DetectorOpenCV::postProcess(cv::Mat& frame, const std::vector<cv::Mat>& outputs, std::vector<BoundingBox> &filtered_boxes)
187{
188 std::vector<int> classIds;
189 std::vector<float> confidences;
190 std::vector<cv::Rect> boxes;
191
192 // Debug: this should be three because there are three scales that Yolo searches over
193 //std::cout << "Outputs: " << outputs.size() << std::endl;
194
195 for (size_t i = 0; i < outputs.size(); ++i)
196 {
197 // Scan through all the bounding boxes output from the network and keep only the
198 // ones with high confidence scores. Assign the box's class label as the class
199 // with the highest score for the box.
200 float* data = reinterpret_cast<float*>(outputs[i].data);
201 for (int j = 0; j < outputs[i].rows; ++j, data += outputs[i].cols)
202 {
203 cv::Mat scores = outputs[i].row(j).colRange(5, outputs[i].cols);
204 cv::Point classIdPoint;
205 double confidence;
206 // Get the value and location of the maximum score
207 minMaxLoc(scores, nullptr, &confidence, nullptr, &classIdPoint);
208
209 if (confidence > 0)
210 {
211
212 // Output is a percentage of the frame width/height
213 // so it doesn't matter that we're transforming boxes
214 // between the resized and full-size image.
215 int centerX = static_cast<int>(data[0] * frame.cols);
216 int centerY = static_cast<int>(data[1] * frame.rows);
217 int width = static_cast<int>(data[2] * frame.cols);
218 int height = static_cast<int>(data[3] * frame.rows);
219 int left = centerX - width / 2;
220 int top = centerY - height / 2;
221
222 classIds.push_back(classIdPoint.x);
223 confidences.push_back(static_cast<float>(confidence));
224 boxes.push_back(cv::Rect(left, top, width, height));
225
226 }else{
227 if(confidence == 0.0)
228 continue;
229
230 std::cout << "Detected "
231 << class_names.at(static_cast<size_t>(classIdPoint.x))
232 << " with low confidence: " << confidence << std::endl;
233
234 }
235 }
236 }
237
238 std::vector<int> indices;
239
240 // Perform non maximum suppression to eliminate redundant overlapping boxes with
241 // lower confidences
242
243 // We set the confidence threshold to zero here, we'll filter the boxes out later.
244 // This lets us provide some feedback to the user if their threshold is too high.
245 cv::dnn::NMSBoxes(boxes, confidences, static_cast<float>(confThreshold), static_cast<float>(nmsThreshold), indices);
246
247 for (size_t i = 0; i < indices.size(); ++i)
248 {
249 auto idx = static_cast<size_t>(indices.at(i));
250
251 BoundingBox box;
252 cv::Rect rect = boxes.at(idx);
253
254 box.confidence = static_cast<double>(confidences.at(idx));
255 box.classid = classIds.at(idx);
256 box.classname = QString::fromStdString(class_names.at(static_cast<size_t>(box.classid)));
257
258 // Darknet predicts box centres and half-width/height, so the
259 // box can go outside the image. Clamp to the image size:
260 QPoint top_left = {std::max(0, rect.x), std::max(0, rect.y)};
261 QPoint bottom_right = top_left + QPoint({rect.width, rect.height});
262
263 bottom_right.setX(std::min(bottom_right.x(), frame.cols));
264 bottom_right.setY(std::min(bottom_right.y(), frame.rows));
265
266 box.rect.setBottomRight(bottom_right);
267 box.rect.setTopLeft(top_left);
268
269 /*
270 std::cout << "Found " << box.classname.toStdString()
271 << " at" << " (" << box.rect.center().x() << ", " << box.rect.center().y()
272 << "), conf: " << box.confidence
273 << ", size (" << box.rect.width() << "x" << box.rect.height() << ")"
274 << std::endl;
275 */
276
277 filtered_boxes.push_back(box);
278
279 }
280
281 return;
282}

284std::vector<BoundingBox> DetectorOpenCV::infer(cv::Mat image){
285
286 std::vector<BoundingBox> detections;
287 running = true;
288
289 // Assume we have an alpha image if 4 channels
290 if(image.channels() == 4){
291 cv::cvtColor(image, image, cv::COLOR_BGRA2BGR);
292 }
293
294 if(convert_depth && image.elemSize() == 2){
295 double minval, maxval;
296 cv::minMaxIdx(image, &minval, &maxval);
297
298 double range = maxval-minval;
299 double scale_factor = 255.0/range;
300
301 image.convertTo(image, CV_32FC1);
302 image -= minval;
303 image *= scale_factor;
304 image.convertTo(image, CV_8UC1);
305 }
306
307 if(convert_grayscale && image.channels() == 1){
308 cv::cvtColor(image, image, cv::COLOR_GRAY2RGB);
309 }
310
311 if(image.channels() != input_channels){
312 std::cout << "Input channel mismatch. Expecting "
313 << input_channels
314 << " channels but image has " << image.channels()
315 << " channels.";
316
317 return detections;
318 }
319
320 // Make sure we service the GUI thread as well, as
321 // this thread will run in a very tight loop
322 QCoreApplication::processEvents();
323 if(framework == FRAMEWORK_TENSORFLOW){
324 detections = inferTensorflow(image);
325 }else if(framework == FRAMEWORK_DARKNET){
326 detections = inferDarknet(image);
327 }
328 QCoreApplication::processEvents();
329
330 running = false;
331
332 return detections;
333

336std::vector<BoundingBox> DetectorOpenCV::inferTensorflow(cv::Mat image){
337
338 std::vector<BoundingBox> results;
339
340 auto mean = cv::Scalar(0,0,0);
341 if(image.channels() == 1){
342 mean = cv::Scalar(0);
343 }
344
345 /*
346 // Check for 16-bit
347 double scale_factor = 1/255.0;
348
349 if(image.elemSize() == 2){
350 scale_factor = 1/65535.0;
351 }
352 */
353
354 auto input_size = cv::Size(image.cols, image.rows);
355
356 bool swap_rb = false; // BGR->RGB?
357 bool crop = false; // Use the entire image
358
359 // No normalising! The model will handle it.
360 auto blob = cv::dnn::blobFromImage(image, 1.0, input_size, mean, swap_rb, crop);
361
362 //Sets the input to the network
363 net.setInput(blob);
364
365 // Runs the forward pass to get output of the output layers
366 std::vector<cv::Mat> outputs;
367 net.forward(outputs, output_names);
368
369 postProcessTensorflow(image, outputs, results);
370
371 std::vector<double> layersTimes;
372 double freq = cv::getTickFrequency() / 1000;
373 processing_time = net.getPerfProfile(layersTimes) / freq;
374
375 return results;
376}

378std::vector<BoundingBox> DetectorOpenCV::inferDarknet(cv::Mat image){
379
380 std::vector<BoundingBox> results;
381
382 auto mean = cv::Scalar(0,0,0);
383 if(image.channels() == 1){
384 mean = cv::Scalar(0);
385 }
386
387 // Check for 16-bit
388 double scale_factor = 1/255.0;
389 if(image.elemSize() == 2){
390 scale_factor = 1/65535.0;
391 }
392
393 auto input_size = cv::Size(input_width, input_height);
394
395 bool swap_rb = true; // BGR->RGB?
396 bool crop = false; // Use the entire image
397 auto blob = cv::dnn::blobFromImage(image, scale_factor, input_size, mean, swap_rb, crop);
398
399 //Sets the input to the network
400 net.setInput(blob);
401
402 // Runs the forward pass to get output of the output layers
403 std::vector<cv::Mat> outputs;
404 net.forward(outputs, output_names);
405
406 // Put efficiency information. The function getPerfProfile returns the
407 // overall time for inference(t) and the timings for each of the layers(in layersTimes)
408 std::vector<double> layersTimes;
409 double freq = cv::getTickFrequency() / 1000;
410 processing_time = net.getPerfProfile(layersTimes) / freq;
411
412 // Remove the bounding boxes with low confidence
413 postProcess(image, outputs, results);
414
415 return results;
416}
