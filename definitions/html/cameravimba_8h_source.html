22 void FrameReceived(const FramePtr pFrame){
23 bool bQueueDirectly = true;
24 VmbFrameStatusType eReceiveStatus;
25 auto receive_error = pFrame->GetReceiveStatus( eReceiveStatus );
26 int num_recievers = receivers(SIGNAL(frameReady(int)));
27
28 if( num_recievers!= 0 && receive_error == VmbErrorSuccess ){
29 // Lock the frame queue
30 frame_mutex.lock();
31 frame_queue.push( pFrame );
32 // Unlock frame queue
33 frame_mutex.unlock();

38 }else{

43 FramePtr getFrame(void){
44 // Lock the frame queue
45 frame_mutex.lock();
46 // Pop frame from queue
47 FramePtr res;
48 if( !frame_queue.empty() )

50 res = frame_queue.front();
51 frame_queue.pop();
52 }
53 // Unlock frame queue
54 frame_mutex.unlock();

56 }
