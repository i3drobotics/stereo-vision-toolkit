190 void AbstractStereoCamera::saveRGBD(QString filename, bool enable_16bit, float scale_16bit){
191  cv::Mat rgbd, color, depth, Q, depth_z, depth_split[3];;
192  getLeftMatchImage(color);
193  getDepth(depth);
194  getQ(Q);
195 
196  // extract Z only channel from depth (xyz) image
197  cv::split(depth, depth_split);
198  depth_z = depth_split[2];
199  // get horizontal fov from Q matrix
200  float hfov = CVSupport::getHFOVFromQ(Q);
201  // embed horizontal fov in top left pixel of z only depth image to simplify reconstruction
202  depth_z.at<float>(0,0) = (float)hfov;
203  if (!depth_z.empty() && !color.empty()){
204  // convert image to rgb
205  if (color.type() == CV_8UC1){
206  cvtColor(color,color,cv::COLOR_GRAY2RGB);
207  } else if (color.type() == CV_8UC3){
208  cvtColor(color,color,cv::COLOR_BGR2RGB);
209  } else {
210  std::cerr << "Invalid image type for saving RGBD" << std::endl;
211  }
212  // Create RGBD image from color image and z only depth image
213  if (enable_16bit){
214  rgbd = CVSupport::createRGBD16(color,depth_z,scale_16bit,false);
215  } else {
216  rgbd = CVSupport::createRGBD32(color,depth_z);
217  }
218  qDebug() << "Saving rgbd image...";
219  cv::imwrite(filename.toStdString(), rgbd);
220  } else {
221  std::cerr << "Depth image or camera image is empty." << std::endl;
222  return;
223  }
224 }
