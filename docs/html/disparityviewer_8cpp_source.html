132 void DisparityViewer::updateDisparity() {
133  cv::Mat disparity;
134  matcher->getDisparity16(disparity);
135  cv::Mat disparity_thresh;
136 
137  disparity.copyTo(disparity_thresh);
138 
139  double min_disp = 10000;
140  double max_disp = 0;
141  double min_depth = 10000;
142  double max_depth = 0;
143 
144  cv::Matx44d _Q;
145  Q.convertTo(_Q, CV_64F);
146 
147  for (int i = 0; i < disparity.rows; i++)
148  {
149  for (int j = 0; j < disparity.cols; j++)
150  {
151  float d = disparity.at<float>(i, j);
152  if (d < 10000 && d > 0){
153  float d = disparity.at<float>(i, j);
154  if (d < min_disp){
155  min_disp = d;
156  max_depth = (double)genZ(_Q,(double)j,(double)i,(double)d);
157  }
158  if (d > max_disp){
159  max_disp = d;
160  min_depth = (double)genZ(_Q,(double)j,(double)i,(double)d);
161  }
162  }
163  }
164  }
165 
166  for (int i = 0; i < disparity.rows; i++)
167  {
168  for (int j = 0; j < disparity.cols; j++)
169  {
170  float d = disparity.at<float>(i, j);
171  if (d > max_disp || d < min_disp){
172  disparity_thresh.at<float>(i, j) = 0;
173  }
174  }
175  }
176 
177  //use depth rather than disparity for more user readable results
178  double range_disp = max_disp - min_disp;
179  double range_depth = max_depth - min_depth;
180 
181  min_disp_ = min_disp;
182  max_disp_ = max_disp;
183  min_depth_ = min_depth;
184  max_depth_ = max_depth;
185 
186  ui->minDisparityLabel->setText(
187  QString("%1 px").arg(QString::number(min_disp,'g', 4)));
188  ui->disparityRangeLabel->setText(
189  QString("%1 px").arg(QString::number(range_disp,'g', 4)));

198  QString("%1 m").arg(QString::number(max_depth,'g', 4)));
199 
200  cv::Mat temp;
201  disparity_thresh.convertTo(temp, CV_32F);
202 
203  cv::normalize(temp, temp, 0, 255, cv::NORM_MINMAX);
204 
205  // Then shift to threshold minimimum displayed disparity
206  //temp -= min_disparity;
207  //cv::threshold(temp, temp, range_disp, 0, cv::THRESH_TRUNC);
208

210 
211  //double scale_factor = 255.0 / (double)range_disp;

218  // cv::ximgproc::getDisparityVis(temp, disparity_scale, scale_factor);

221 
222  //colour_disparity = disparity_vis;
223 
224  cv::cvtColor(disparity_vis, disparity_vis, CV_BGR2RGB);
225 
226  auto left_ptr = matcher->getLeftImage();
227 
228  for (int x = 0; x < disparity_scale.cols; x++) {
229  for (int y = 0; y < disparity_scale.rows; y++) {
230  if (disparity_scale.at<uchar>(y, x) == 255 ||
231  disparity_scale.at<uchar>(y, x) == 0 ||

244  disparity_vis.at<cv::Vec3b>(y, x)[0] = 0;
245  disparity_vis.at<cv::Vec3b>(y, x)[1] = 0;
246  disparity_vis.at<cv::Vec3b>(y, x)[2] = 0;
247  }
248 
249  }
250  }
251 
252  colour_disparity = disparity_vis;
253 
254  QImage dmap(disparity_vis.data, disparity_vis.cols, disparity_vis.rows,
255  QImage::Format_RGB888);

257  QPixmap pmap_disparity = QPixmap::fromImage(dmap);
258 
259  if(!pmap_disparity.isNull()){
260  emit newDisparity(
