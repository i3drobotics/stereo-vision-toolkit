76 void AbstractStereoCamera::reproject3D() {
77 
78  cv::Mat disparity_downscale;
79 
80  matcher->getDisparity(disparity_downscale);
81 
82  cv::abs(disparity_downscale);
83 
84  // By default the disparity maps are scaled by a factor of 16.0
85  disparity_downscale /= 16.0;
86 
87  if (Q.empty() || disparity_downscale.empty()) {
88  return;
89  }
90 
91  cv::reprojectImageTo3D(disparity_downscale, stereo_reprojected, Q, true);
92 
93  pcl::PointCloud<pcl::PointXYZRGB>::Ptr ptCloudTemp(
94  new pcl::PointCloud<pcl::PointXYZRGB>);
95 
96  pcl::PointXYZRGB point;
97  uint32_t rgb = 0;
98  uchar col = 0;
99 
100  point.x = 0;
101  point.y = 0;
102  point.z = 0;
103 
104  rgb = ((int)255) << 16 | ((int)255) << 8 | ((int)255);
105  point.rgb = *reinterpret_cast<float *>(&rgb);
106  ptCloudTemp->points.push_back(point);
107 
108  for (int i = 0; i < stereo_reprojected.rows; i++) {
109  float *reconst_ptr = stereo_reprojected.ptr<float>(i);
110  float *disp_ptr = matcher->disparity_lr.ptr<float>(i);
111  uchar *rgb_ptr = left_remapped.ptr<uchar>(i);
112 
113  // TODO: Figure out why this can trigger, assume the disparity map isn't generated
114  // when this function gets called.
115  if(!disp_ptr || !rgb_ptr || !reconst_ptr) return;
116 
117  for (int j = 0; j < stereo_reprojected.cols; j++) {
118  if (disp_ptr[j] == 0) continue;
119  if (rgb_ptr[j] == 0) continue;
120 
121  point.x = reconst_ptr[3 * j];
122  point.y = -reconst_ptr[3 * j + 1];
123  point.z = -reconst_ptr[3 * j + 2];
124 
125  if(abs(point.x) > 10) continue;
126  if(abs(point.y) > 10) continue;
127  if(point.z == -10000) continue;
128  col = rgb_ptr[j];
129 
130  rgb = ((int)col) << 16 | ((int)col) << 8 | ((int)col);
131  point.rgb = *reinterpret_cast<float *>(&rgb);
132 
133  ptCloudTemp->points.push_back(point);
134  }
135  }
136 
137  if(ptCloudTemp->points.empty())
138  return;
139 
140  pcl::PassThrough<pcl::PointXYZRGB> pass;
141  pass.setInputCloud (ptCloudTemp);
142  pass.setFilterFieldName ("z");
143  pass.setFilterLimits(-visualisation_max_z, -visualisation_min_z);
144  pass.filter(*ptCloudTemp);
145 
146  /*
147  pcl::RadiusOutlierRemoval<pcl::PointXYZRGB> outrem;
148  outrem.setInputCloud(ptCloudTemp);
149  outrem.setRadiusSearch(0.05);
150  outrem.setMinNeighborsInRadius (5);
151  outrem.filter (*ptCloudTemp);
152 
153  pcl::StatisticalOutlierRemoval<pcl::PointXYZRGB> sor;
154  sor.setInputCloud (ptCloudTemp);
155  sor.setMeanK (5);
156  sor.setStddevMulThresh (4.0);
157  sor.filter(*ptCloudTemp);
158  */
159 
160  ptCloud = ptCloudTemp;
161 
162  emit reprojected();
163 }

383 void AbstractStereoCamera::captureAndProcess(void) {
384  QElapsedTimer frametimer;
385  frametimer.restart();
386 
387  capturing = true;
388 
389  if (capture()) {
390  frames++;
391  emit framecount(frames);
392 
393  if (rectifying) {
394  rectifyImages();
395  } else {
396  left_raw.copyTo(left_remapped);
397  right_raw.copyTo(right_remapped);
398  }
399 
400  left_remapped.copyTo(left_output);
401  right_remapped.copyTo(right_output);
402 
403  if (matching) {
404  matcher->match();
405  if (reprojecting) {
406  reproject3D();
407  }
408  emit matched();
409  }
410 
411  emit acquired();
412  emit fps(frametimer.elapsed());
413  frametimer.restart();
414  }
415 
416  capturing = false;
417 }
