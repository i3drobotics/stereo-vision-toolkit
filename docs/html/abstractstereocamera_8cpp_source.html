120 void AbstractStereoCamera::reproject3D() {
121 
122  cv::Mat disparity_downscale;
123 
124  matcher->getDisparity(disparity_downscale);
125 
126  cv::abs(disparity_downscale);
127 
128  // By default the disparity maps are scaled by a factor of 16.0
129  disparity_downscale /= 16.0;
130 
131  int row = (disparity_downscale.rows-1)/2;
132  int column = (disparity_downscale.cols-1)/2;
133  float disp_val_f = disparity_downscale.at<float>(row, column);
134  qDebug() << "IMAGE CENTER: " << column+1 << "," << row+1;
135  qDebug() << "DISPARITY AT CENTER: " << disp_val_f;
136 
137  if (Q.empty() || disparity_downscale.empty()) {
138  return;
139  }
140 
141  cv::reprojectImageTo3D(disparity_downscale, stereo_reprojected, Q, true);
142 
143  pcl::PointCloud<pcl::PointXYZRGB>::Ptr ptCloudTemp(
144  new pcl::PointCloud<pcl::PointXYZRGB>);
145 
146  pcl::PointXYZRGB point;
147  uint32_t rgb = 0;
148  uchar col = 0;
149 
150  point.x = 0;
151  point.y = 0;
152  point.z = 0;
153 
154  rgb = ((int)255) << 16 | ((int)255) << 8 | ((int)255);
155  point.rgb = *reinterpret_cast<float *>(&rgb);
156  ptCloudTemp->points.push_back(point);
157 
158  for (int i = 0; i < stereo_reprojected.rows; i++) {
159  float *reconst_ptr = stereo_reprojected.ptr<float>(i);
160  float *disp_ptr = matcher->disparity_lr.ptr<float>(i);
161  uchar *rgb_ptr = left_remapped.ptr<uchar>(i);
162 
163  // TODO: Figure out why this can trigger, assume the disparity map isn't generated
164  // when this function gets called.
165  if(!disp_ptr || !rgb_ptr || !reconst_ptr) return;
166 
167  for (int j = 0; j < stereo_reprojected.cols; j++) {
168  if (disp_ptr[j] == 0) continue;
169  if (rgb_ptr[j] == 0) continue;
170 
171  if (j == column){
172  if (i == row){
173  float reproj_val_2 = reconst_ptr[3 * j + 2];
174  qDebug() << "REPROJECT3D AT CENTER 2: " << reproj_val_2;
175  }
176  }
177 
178  point.x = -reconst_ptr[3 * j];
179  point.y = -reconst_ptr[3 * j + 1];
180  point.z = reconst_ptr[3 * j + 2];
181 
182  if(abs(point.x) > 10) continue;
183  if(abs(point.y) > 10) continue;
184  if(point.z == 10000) continue;
185  if(point.z == -10000) continue;
186  col = rgb_ptr[j];
187 
188  rgb = ((int)col) << 16 | ((int)col) << 8 | ((int)col);
189  point.rgb = *reinterpret_cast<float *>(&rgb);
190 
191  ptCloudTemp->points.push_back(point);
192  }
193  }
194 
195  if(ptCloudTemp->points.empty())
196  return;
197 
198  pcl::PassThrough<pcl::PointXYZRGB> pass;
199  pass.setInputCloud (ptCloudTemp);
200  pass.setFilterFieldName ("z");
201  pass.setFilterLimits(visualisation_min_z, visualisation_max_z);
202  pass.filter(*ptCloudTemp);
203 
204  /*
205  pcl::RadiusOutlierRemoval<pcl::PointXYZRGB> outrem;
206  outrem.setInputCloud(ptCloudTemp);
207  outrem.setRadiusSearch(0.05);
208  outrem.setMinNeighborsInRadius (5);
209  outrem.filter (*ptCloudTemp);
210 
211  pcl::StatisticalOutlierRemoval<pcl::PointXYZRGB> sor;
212  sor.setInputCloud (ptCloudTemp);
213  sor.setMeanK (5);
214  sor.setStddevMulThresh (4.0);
215  sor.filter(*ptCloudTemp);
216  */
217 
218  ptCloud = ptCloudTemp;
219 
220  emit reprojected();
221 }
