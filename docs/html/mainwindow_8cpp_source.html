405 void MainWindow::runCalibrationFromImages(void){
406 
407  qDebug() << "Beginning calibration from images";
408 
409  if(calibration_images_dialog == NULL) return;
410  qDebug() << "Getting parameters";
411 
412  int cols = calibration_images_dialog->getPatternCols();
413  int rows = calibration_images_dialog->getPatternRows();
414  double square_size_mm = calibration_images_dialog->getSquareSizeMm();
415  auto left_images = calibration_images_dialog->getLeftImages();
416  auto right_images = calibration_images_dialog->getRightImages();
417 
418  calibration_images_dialog->close();
419 
420  calibrator = new StereoCalibrate(this, NULL);
